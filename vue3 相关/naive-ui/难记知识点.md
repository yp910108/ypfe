## naive-ui@2.34.3

#### Date Picker

```html
<template>
  <n-space vertical>
    禁用上半月
    <n-date-picker
      type="date"
      :default-value="Date.now()"
      :is-date-disabled="dateDisabled"
    />
    禁用上半月 & 下半个月的上午
    <n-date-picker
      type="datetime"
      :default-value="Date.now()"
      :is-date-disabled="dateDisabled"
      :is-time-disabled="timeDisabled"
    />
  </n-space>
</template>

<script lang="ts">
import { defineComponent } from 'vue'

export default defineComponent({
  setup() {
    return {
      dateDisabled(ts: number) {
        const date = new Date(ts).getDate()
        return date < 15
      },
      timeDisabled(ts: number) {
        const date = new Date(ts).getDate()
        return {
          isHourDisabled: (hour: number) => {
            return date >= 15 && hour < 12
          }
        }
      }
    }
  }
})
</script>
```

```html
<template>
  <n-space vertical>
    间隔至少七天
    <n-date-picker
      type="daterange"
      :default-value="[1675155238634, 1675155238634 + 7 * 86400000]"
      :is-date-disabled="isDateDisabled"
    />
    间隔至少七天
    <n-date-picker
      type="datetimerange"
      :default-value="[1675155231634, 1675155231634 + 7 * 86400000]"
      :is-date-disabled="isDateDisabled"
      :is-time-disabled="isTimeDisabled"
      :default-time="['16:53:58', '16:53:58']"
    />
  </n-space>
</template>

<script lang="ts">
import { defineComponent } from 'vue'
import { startOfDay } from 'date-fns/esm'

const d = 86400000
const h = 3600000
const m = 60000
const s = 1000

export default defineComponent({
  setup() {
    return {
      isDateDisabled(
        date: number,
        position: 'start' | 'end',
        value: [number, number] | null
      ) {
        if (value && value.length) {
          const distance =
            position === 'start' ? value[1] - date : date - value[0]
          return distance <= 6 * d
        } else {
          return false
        }
      },
      isTimeDisabled(
        date: number,
        position: 'start' | 'end',
        value: [number, number]
      ) {
        const [start, end] = value
        if (position === 'start') {
          return {
            isHourDisabled: (hour: number) => {
              return end - (startOfDay(start).valueOf() + h * hour) < 7 * d
            },
            isMinuteDisabled: (minute: number, hour: number | null) => {
              return (
                end - (startOfDay(start).valueOf() + h * hour! + m * minute) <
                7 * d
              )
            },
            isSecondDisabled: (
              second: number,
              minute: number | null,
              hour: number | null
            ) => {
              return (
                end -
                  (startOfDay(start).valueOf() +
                    h * hour! +
                    m * minute! +
                    s * second) <
                7 * d
              )
            }
          }
        } else {
          return {
            isHourDisabled: (hour: number) => {
              return startOfDay(end).valueOf() + h * (hour + 1) - start < 7 * d
            },
            isMinuteDisabled: (minute: number, hour: number | null) => {
              return (
                startOfDay(end).valueOf() +
                  h * hour! +
                  m * (minute + 1) -
                  start <
                7 * d
              )
            },
            isSecondDisabled: (
              second: number,
              minute: number | null,
              hour: number | null
            ) => {
              return (
                startOfDay(end).valueOf() +
                  h * hour! +
                  m * minute! +
                  s * (second + 1) -
                  start <
                7 * d
              )
            }
          }
        }
      }
    }
  }
})
</script>
```

#### Data Table

只要有一列的 `sortOrder` 被设置，表格的排序便成为受控模式：

```html
<template>
  <n-data-table
    :columns="columns"
    :data="data"
    :pagination="pagination"
    @update:sorter="handleSorterChange"
  />
</template>

<script>
import { defineComponent, ref } from 'vue'

const columns = [
  {
    title: 'Name',
    key: 'name',
    sortOrder: false,
    sorter: 'default'
  },
  {
    title: 'Age',
    key: 'age',
    sortOrder: false,
    sorter: {
      multiple: 1,
      compare: (rowA, rowB) => {
        return rowA.age - rowB.age
      }
    }
  },
  {
    title: 'Address',
    key: 'address'
  }
]

const data = [
  {
    key: 0,
    name: 'John Brown',
    age: 38,
    address: 'New York No. 1 Lake Park'
  },
  {
    key: 1,
    name: 'Jim Green',
    age: 42,
    address: 'London No. 1 Lake Park'
  },
  {
    key: 2,
    name: 'Joe Black',
    age: 36,
    address: 'Sidney No. 1 Lake Park'
  },
  {
    key: 3,
    name: 'Jim Red',
    age: 32,
    address: 'London No. 2 Lake Park'
  }
]

export default defineComponent({
  setup() {
    const columnsRef = ref(columns)
    return {
      data,
      columns: columnsRef,
      pagination: { pageSize: 5 },
      handleSorterChange(sorter) {
        console.log(sorter)
        if (Array.isArray(sorter)) {
          sorter = sorter[0]
        }
        for (const column of columnsRef.value) {
          if (column.sortOrder !== undefined) {
            if (column.key === sorter.columnKey) {
              column.sortOrder = sorter.order
            } else {
              column.sortOrder = false
            }
          }
        }
      }
    }
  }
})
</script>
```

哪一列设置 `filterOptionValues`（设置为 `null` 为受控，设置为 `undefined` 为非受控），则哪一列成为受控模式：

```html
<template>
  <n-data-table
    ref="table"
    :columns="columns"
    :data="data"
    :pagination="pagination"
    @update:filters="handleUpdateFilter"
  />
</template>

<script>
import { defineComponent, ref } from 'vue'

const data = [
  {
    key: 0,
    name: 'John Brown',
    age: 32,
    address: 'New York No. 1 Lake Park'
  },
  {
    key: 1,
    name: 'Jim Green',
    age: 42,
    address: 'London No. 1 Lake Park'
  },
  {
    key: 2,
    name: 'Joe Black',
    age: 32,
    address: 'Sidney No. 1 Lake Park'
  },
  {
    key: 3,
    name: 'Jim Red',
    age: 32,
    address: 'London No. 2 Lake Park'
  }
]

export default defineComponent({
  setup() {
    const tableRef = ref(null)

    const columnsRef = ref([
      {
        title: 'Name',
        key: 'name'
      },
      {
        title: 'Age',
        key: 'age',
        filterOptionValues: [32],
        filterOptions: [
          {
            label: 32,
            value: 32
          },
          {
            label: 42,
            value: 42
          }
        ],
        filter(value, row) {
          return row.age === value
        }
      },
      {
        title: 'Address',
        key: 'address',
        filterOptionValues: ['London'],
        filterOptions: [
          {
            label: 'London',
            value: 'London'
          },
          {
            label: 'New York',
            value: 'New York'
          }
        ],
        filter(value, row) {
          return row.address.includes(value)
        }
      }
    ])

    return {
      table: tableRef,
      data,
      columns: columnsRef,
      pagination: { pageSize: 5 },
      handleUpdateFilter(filters, sourceColumn) {
        const column = columnsRef.value.find(
          ({ key }) => key === sourceColumn.key
        )
        column.filterOptionValues = filters[sourceColumn.key]
      }
    }
  }
})
</script>
```

异步示例：

```html
<template>
  <n-data-table
    ref="table"
    remote
    :columns="columns"
    :data="data"
    :loading="loading"
    :pagination="pagination"
    @update:sorter="handleSorterChange"
    @update:filters="handleFiltersChange"
    @update:page="handlePageChange"
  />
</template>

<script>
import { defineComponent, ref, onMounted } from 'vue'

const originColumns = [
  {
    title: 'Column1',
    key: 'column1',
    sortOrder: false,
    sorter: true
  },
  {
    title: 'Column2',
    key: 'column2',
    filterOptionValues: [],
    filterOptions: [
      { label: 'column2-1', value: 'column2-1' },
      { label: 'column2-2', value: 'column2-2' }
    ],
    filter: true
  },
  {
    title: 'Column3',
    key: 'column3'
  }
]

const originData = Array.from({ length: 46 }, (_, index) => ({
  // key: index,
  column1: `column1-${index}`,
  column2: `column2-${index}`,
  column3: `column3-${index}`
}))

function fetch(page, pageSize = 10, order = 'ascend', filterValues = []) {
  return new Promise((resolve) => {
    const data = [...originData]
    const orderedData = order === 'descend' ? data.reverse() : data
    const filteredData = filterValues.length
      ? orderedData.filter((row) =>
          filterValues.find((value) => row.column2.includes(value))
        )
      : orderedData
    const total = filteredData.length
    const pagedData = filteredData.slice((page - 1) * pageSize, page * pageSize)
    setTimeout(() => resolve({ total, data: pagedData }), 500)
  })
}

export default defineComponent({
  setup() {
    const loading = ref(true)
    const columns = ref(originColumns)
    const data = ref([])
    const pagination = ref({
      page: 1,
      pageSize: 10,
      itemCount: 0,
      // pageCount: 1,
      prefix({ itemCount }) {
        return `Total is ${itemCount}.`
      }
    })

    onMounted(() => {
      fetch(
        pagination.value.page,
        pagination.value.pageSize,
        columns.value[0].sortOrder,
        columns.value[1].filterOptionValues
      ).then(({ total, data: _data }) => {
        data.value = _data
        pagination.value.itemCount = total
        // pagination.value.pageCount = Math.ceil(
        //   total / pagination.value.pageSize
        // )
        loading.value = false
      })
    })

    return {
      loading,
      columns,
      data,
      pagination,
      handleSorterChange(sorter) {
        if (!sorter || sorter.columnKey === 'column1') {
          if (!loading.value) {
            loading.value = true
            fetch(
              pagination.value.page,
              pagination.value.pageSize,
              !sorter ? false : sorter.order,
              columns.value[1].filterOptionValues
            ).then(({ total, data: _data }) => {
              columns.value[0].sortOrder = !sorter ? false : sorter.order
              data.value = _data
              pagination.value.itemCount = total
              // pagination.value.pageCount = Math.ceil(
              //   total / pagination.value.pageSize
              // )
              loading.value = false
            })
          }
        }
      },
      handleFiltersChange(filters) {
        if (!loading.value) {
          loading.value = true
          const filterValues = filters.column2 || []
          fetch(
            pagination.value.page,
            pagination.value.pageSize,
            columns.value[0].sortOrder,
            filterValues
          ).then(({ total, data: _data }) => {
            columns.value[1].filterOptionValues = filterValues
            data.value = _data
            pagination.value.itemCount = total
            // pagination.value.pageCount = Math.ceil(
            //   total / pagination.value.pageSize
            // )
            loading.value = false
          })
        }
      },
      handlePageChange(currentPage) {
        if (!loading.value) {
          loading.value = true
          fetch(
            currentPage,
            pagination.value.pageSize,
            columns.value[0].sortOrder,
            columns.value[1].filterOptionValues
          ).then(({ total, data: _data }) => {
            data.value = _data
            pagination.value.page = currentPage
            pagination.value.itemCount = total
            // pagination.value.pageCount = Math.ceil(
            //   total / pagination.value.pageSize
            // )
            loading.value = false
          })
        }
      }
    }
  }
})
</script>
```

排序注意，数据很多时，sorter 不能指定方法，内置的排序功能会很卡。

`render-expand-icon` 、`render-cell` 需要在 <NDataTable /> 上面声明，其他的 render 需要在 DataTableColumn 中声明。
